  while (reactorCoreTemp < maxTemp) {
    reactorCoreTemp += Math.random() * 10;
    if (cooling = false) {
      reactorCoreTemp -= 50;
    }
    setTimeout(() => console.log("Core temp:", reactorCoreTemp), 1000);
  }
  alert("Critical temperature reached! Initiate meltdown protocol.");
}
